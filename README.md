# Install Files and Script

apt update && apt upgrade -y;

pkg install python2 -y;

pkg install git -y;

git clone https://github.com/ThonyDroidYT/TD-CCGen

cd

ls

python2 TD-CCGen.py

python2 TD-CCGen.py -b 483039xxxxxxxxxx -u 20 -d -c
